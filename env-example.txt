# ===========================================
# DATABASE (Required) — choose ONE of the modes below
# ===========================================
#
# The connection is resolved in lib/db-url.ts in this precedence order:
#   1. PGHOST + PGPASSWORD  -> static-password connection string (local/dev)
#   2. DATABASE_URL         -> explicit connection string
#   3. PGHOST + AWS_ROLE_ARN (and NO PGPASSWORD/DATABASE_URL) -> RDS IAM auth
#
# --- Mode A: static connection string (local Docker / simple hosted PG) ---
DATABASE_URL="postgresql://USER:PASSWORD@HOST:PORT/DB_NAME?sslmode=require"
# Vercel also adds these which Prisma can use:
# POSTGRES_URL=
# POSTGRES_PRISMA_URL=
# POSTGRES_URL_NON_POOLING=
#
# --- Mode B: RDS IAM database auth via Vercel OIDC (recommended for prod) ---
# Leave PGPASSWORD *and* DATABASE_URL unset so the app mints a short-lived IAM
# token per connection (no static DB password stored anywhere). Requires:
#   - RDS instance with IAM database authentication enabled
#   - DB user GRANTed rds_iam
#   - A Vercel<->AWS OIDC role with the rds-db:connect permission
# PGHOST="your-instance.xxxx.us-east-1.rds.amazonaws.com"
# PGPORT="5432"
# PGUSER="app_user"
# PGDATABASE="postgres"
# AWS_REGION="us-east-1"
# AWS_ROLE_ARN="arn:aws:iam::ACCOUNT_ID:role/your-vercel-oidc-role"
# Optional: cap pool size (RDS connection limits are low). Defaults to 20.
# PG_POOL_MAX="20"
#
# NOTE: The Prisma CLI (migrate/seed) cannot use Vercel OIDC. For migrations
# against an IAM-only RDS instance, run them with a temporary admin
# DATABASE_URL (or from inside the VPC), not from the IAM path above.
